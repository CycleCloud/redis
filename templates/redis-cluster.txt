[parameters General Settings]
Order = 10

    [[parameters Cloud Service Provider Configuration]]
    Description = Configure the Cloud Provider account options.
    Order = 10

        [[[parameter CloudProvider]]]
        Label = Cloud Provider
        ParameterType = Cloud.Provider

        [[[parameter Credentials]]]
        Description = The credentials for the cloud provider
        ParameterType = Cloud.Credentials

        [[[parameter Region]]]
        Label = Location
        Description = Deployment Location
        ParameterType = Cloud.Region

    [[parameters Node Access]]
    Description = "Specify the credentials required for access to the node.  The Keypair Path specifies the path on local machines for cyclecloud CLI access."
    Order = 20

        [[[parameter keypairLocation]]]
        Label = Keypair Path
        Description = The path to the keypair
        DefaultValue = ~/.ssh/cyclecloud.pem
        Required = True

        [[[parameter Username]]]
        Label = Username
        Description = The username for the built-in local user on all nodes for direct ssh access.
        DefaultValue = cyclecloud

        [[[parameter Password]]]
        Label = Password
        ParameterType = Password
        Description = The password for the built-in local user on all nodes for direct ssh access.
        Conditions.Required := startswith("cycle.image.win", ImageName)
        Conditions.Excluded := !startswith("cycle.image.win", ImageName)


[parameters Cluster Software]
Order = 30
Description = Specify the scheduling software, and base OS installed on all nodes, and optionally the cluster-init and chef versions from your Locker.

    [[parameters Software]]
    Order = 10
    
        [[[parameter ImageName]]]
        Label = Base OS
        ParameterType = Cloud.Image
        Config.OS = linux
        #        DefaultValue = cycle.image.centos7
        DefaultValue = cycle.image.ubuntu14

        [[[parameter ProjectVersion]]]
        Label = Project Version
        DefaultValue = 1.1.2
        Description = Select the Redis project spec version
        Hidden = true

        [[[parameter StageResources]]]
        Label = Stage Resources
        DefaultValue = true
        Description = whether or not to stage resources
        Hidden = true

    [[parameters Redis]]
    Description = Adjust the Redis configuration options.  (Data redunancy and fault tolerance requires more than 1 server (max: 10).)
    Order = 20

        [[[parameter RedisClusterSize]]]
        Label = Cluster Size
        Description = The number of Redis Servers to start initially.
        DefaultValue = 1
        Config.Plugin = pico.form.NumberTextBox
        Config.MinValue = 1
        Config.MaxValue = 10
        Config.IntegerOnly = true


[parameters Compute Backend]
    Order = 40

    [[parameters Instance Types]]
    Description = "The cluster, in this case, has two roles: the return proxy node and the Redis server hosts. Configure which instance types to use based on the requirements of your application.  The proxy host may be used to enable firewall traversal between the cluster and CycleCloud.   In VPC environments with virtual network, this node is not required."
    Order = 10

        [[[parameter ProxyMachineType]]]
        Label = Proxy
        Description = The machine type for the optional proxy host
        ParameterType = Cloud.MachineType
        DefaultValue = Standard_A2

       [[[parameter ServerMachineType]]]
       Label = Server
       Description = The machine type for the Redis servers
       ParameterType = Cloud.MachineType
       DefaultValue = Standard_DS4_v2 

    [[parameter EnableTerminationProtection]]
    Label = Term Protection
    DefaultValue = false
    Widget.Plugin = pico.form.BooleanCheckBox
    Widget.Label = Enable Termination Protection?


[parameters Networking]
Order = 50

    [[parameters Azure Virtual Network Configuration]]
    Order = 40
    Conditions.Excluded := CloudProvider !== "Azure"

        [[[parameter AzureSubnetId]]]
        Label = Subnet ID
        Description = Subnet Resource Path (ResourceGroup/VirtualNetwork/Subnet)
        ParameterType = Azure.Subnet
        Required = true
        
    [[parameter ExecuteNodesPublic]]
    Label = Public Servers
    DefaultValue = false
    Widget.Plugin = pico.form.BooleanCheckBox
    Widget.Label = Make the Redis servers accessible from the Internet

    
[cluster RedisCluster]
FormLayout = selectionpanel
IconUrl = http://download.cyclecomputing.com/logos/redis-192x192.png

     # Autoscale for Redis is out-of-scope for this example
     # If required, please contact CycleComputing for additional support
     Autoscale = false

     [[node defaults]]
     TerminationProtection = $EnableTerminationProtection
     ImageName = $ImageName

     StageResources = $StageResources
     # Chef and Locker Config
     Credentials = $Credentials
     KeyPairLocation = $KeypairLocation
     SubnetId = $AzureSubnetId

     [[[cluster-init redis-cluster:default:$ProjectVersion]]]
     Locker = cyclecloud

     # Azure
     Region = $Region

     [[node proxy]]
     IsReturnProxy = true
     MachineType = $ProxyMachineType
     SubnetId = $AzureSubnetId

         [[[configuration]]]
         run_list = recipe[cganglia::server]
         cyclecloud.discoverable = true


         [[[cluster-init redis-cluster:proxy:$ProjectVersion]]]
         # Cycle Examples Repository
         Locker = cyclecloud

         # Configure Azure external input endpoints (for example SSH)
         [[[input-endpoint SSH]]]
         PrivatePort = 22
         PublicPort = 22

         [[[input-endpoint ganglia]]]
         PrivatePort = 8652
         PublicPort = 8652        


    [[nodearray server]]
    MachineType = $ServerMachineType
    InitialCount = $RedisClusterSize

         [[[network-interface]]]
         AssociatePublicIpAddress = $ExecuteNodesPublic

        [[[configuration]]]
        run_list = recipe[cganglia::client]
        cyclecloud.discoverable = true
        redis.cluster_size = $RedisClusterSize

        [[[cluster-init redis-cluster:server:$ProjectVersion]]]
        Locker = cyclecloud
        # Configure Azure external input endpoints (for example SSH)
        [[[input-endpoint SSH]]]
        PrivatePort = 22
        PublicPort = 55100
